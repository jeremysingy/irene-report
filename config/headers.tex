%------------------------------------------------
% Headers and footers configuration
%------------------------------------------------

% Custom header height
%\setlength{\headheight}{22pt}

% Show headers also in plain pages
%\fancypagestyle{plain}{}

% No uppercase for chapter and section names
\renewcommand{\chaptermark}[1]{\markboth{\bsc{\chaptername~\thechapter{} :} #1}{}}
\renewcommand{\sectionmark}[1]{\markright{\thesection{} #1}}

% Remove page number in plain pages
\fancypagestyle{plain}{\fancyhf{}\renewcommand{\headrulewidth}{0pt}\renewcommand{\footrulewidth}{0pt}}

% Basic use of fancyhdr (without multiple styles)
%\pagestyle{fancy}
%\fancyhf{}
%
%% Add a rule also for footer
%\renewcommand{\footrulewidth}{0.4pt}
%
%% Odd page (recto, right-hand) and even page (verso, left-hand)
%\fancyhead[RE,LO]{\sffamily\doctitle}
%\fancyhead[LE,RO]{\sffamily\nouppercase\rightmark}
%\fancyfoot[RE,LO]{\sffamily\authornamesf}
%\fancyfoot[LE,RO]{\sffamily \thepage{} of \pageref*{LastPage}}

% Page style for main document
\fancypagestyle{mainfancy}
{%
    \fancyhf{}
    
    % Add rules
    \renewcommand{\headrulewidth}{0.4pt}
    \renewcommand{\footrulewidth}{0.4pt}
    
    % Odd pages (recto, right-hand) and even pages (verso, left-hand)
    \fancyhead[RE,LO]{\sffamily\doctitle}
    \fancyhead[LE,RO]{\sffamily\nouppercase\rightmark}
    \fancyfoot[RE,LO]{\sffamily\authornamesf}
    \fancyfoot[LE,RO]{\sffamily \thepage{} of \pageref*{LastPage}}
}

% Commands to simplify header style and numbering changing
\newcommand{\frontmatterstyle}
{
    % Plain style with page number in footer center
    \fancypagestyle{plain}{%
        \fancyhf{}\renewcommand{\headrulewidth}{0pt}\renewcommand{\footrulewidth}{0pt}
        \fancyfoot[C]{\sffamily\thepage}
    }
    \pagestyle{plain}
    \pagenumbering{roman} % Roman numerals
}
\newcommand{\maindocstyle}
{
    % Plain style completely empty (for first page of chapters)
    \fancypagestyle{plain}{%
        \fancyhf{}\renewcommand{\headrulewidth}{0pt}\renewcommand{\footrulewidth}{0pt}
    }
    \pagestyle{mainfancy}
    \pagenumbering{arabic}
}
